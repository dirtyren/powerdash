# Plan 2.1 — Edit Existing Dashboard — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Enable users to drag/resize/reorder/remove widgets on an existing OpMon dashboard and persist the change through the seagull backend.

**Architecture:** New route `/dashboards/[id]/edit` with a `react-grid-layout`-based editable canvas. Local React state tracks in-progress edits; `useSaveDashboard` (TanStack Query `useMutation`) PUTs a JSON payload to `/api/dashboards/[id]`; the route handler calls a new `saveDashboard(id, widgets)` server adapter that serializes a seagull XML body via `buildDashboardXml` and posts it to seagull. Read path from P1 is unchanged.

**Tech Stack:** Next.js 15 App Router, React 19, TypeScript strict, TanStack Query v5, `react-grid-layout` v1, Zod v3, `fast-xml-parser` v4 (existing, for round-trip tests), Vitest v2, Playwright v1, WireMock 3.9.

**Spec:** `docs/superpowers/specs/2026-04-29-plan-2.1-edit-existing-dashboard.md`.

- [ ] **Step 4: Implement `src/server/seagull/xml-build.ts`**

```ts
import type { Dashboard } from "../schemas/dashboard";

const XML_ESCAPES: Record<string, string> = {
  "&": "&amp;",
  "<": "&lt;",
  ">": "&gt;",
  '"': "&quot;",
  "'": "&apos;",
};

function escapeXml(value: string | number): string {
  return String(value).replace(/[&<>"']/g, (c) => XML_ESCAPES[c] ?? c);
}

function widgetXml(w: Dashboard["widgets"][number]): string {
  return [
    "<widget>",
    `<id>${escapeXml(w.id)}</id>`,
    `<kind>${escapeXml(w.kind)}</kind>`,
    `<title>${escapeXml(w.title)}</title>`,
    `<x>${w.x}</x>`,
    `<y>${w.y}</y>`,
    `<w>${w.w}</w>`,
    `<h>${w.h}</h>`,
    "</widget>",
  ].join("");
}

/**
 * Serialize a Dashboard into the XML body seagull accepts on save.
 *
 * The shape mirrors the GET /dashboards/{id}.xml response envelope:
 *   <response><dashboard>…<widgets><widget/>…</widgets></dashboard></response>
 *
 * Numeric fields (x/y/w/h) are emitted as plain numbers — on the read side
 * they arrive as strings and are coerced by z.coerce.number() in the schema.
 */
export function buildDashboardXml(dashboard: Dashboard): string {
  const widgetsXml =
    dashboard.widgets.length === 0
      ? "<widgets/>"
      : `<widgets>${dashboard.widgets.map(widgetXml).join("")}</widgets>`;

  return (
    `<?xml version="1.0" encoding="UTF-8"?>` +
    `<response>` +
    `<dashboard>` +
    `<id>${escapeXml(dashboard.id)}</id>` +
    `<name>${escapeXml(dashboard.name)}</name>` +
    `<owner>${escapeXml(dashboard.owner)}</owner>` +
    widgetsXml +
    `</dashboard>` +
    `</response>`
  );
}
```

```ts
import { z } from "zod";
import { getDashboard, saveDashboard } from "@/server/seagull/dashboards";
import { SeagullError } from "@/server/seagull/client";
import { WidgetRefSchema } from "@/server/schemas/widget";

// ... existing GET handler stays unchanged ...

const PutBodySchema = z.object({
  widgets: z.array(WidgetRefSchema),
});

export async function PUT(req: Request, ctx: { params: Promise<{ id: string }> }) {
  const { id } = await ctx.params;

  let parsedBody: z.infer<typeof PutBodySchema>;
  try {
    const raw: unknown = await req.json();
    parsedBody = PutBodySchema.parse(raw);
  } catch (err) {
    return NextResponse.json(
      {
        error: "invalid_body",
        message: err instanceof Error ? err.message : String(err),
      },
      { status: 422 },
    );
  }

  try {
    const existing = await getDashboard(id);
    const saved = await saveDashboard({
      ...existing,
      widgets: parsedBody.widgets,
    });
    return NextResponse.json(saved, {
      headers: { "Cache-Control": "no-store" },
    });
  } catch (err) {
    if (err instanceof SeagullError) {
      return NextResponse.json(
        { error: "upstream", status: err.status, message: err.message },
        { status: 502 },
      );
    }
    return NextResponse.json(
      {
        error: "internal",
        message: err instanceof Error ? err.message : String(err),
      },
      { status: 500 },
    );
  }
}
```

Note: we fetch the existing dashboard (`getDashboard`) to preserve `name` and `owner` — the edit UI only ships `widgets`. If the discovery in Task 1 revealed seagull can accept a widgets-only patch, simplify this later; the current shape is the safe default.

Ensure the import list at the top of the file includes `saveDashboard`, `WidgetRefSchema`, and `z` — the existing GET imports may already cover `SeagullError`, `NextResponse`, and `getDashboard`. Deduplicate as needed.

```tsx
import { useMemo, useState } from "react";
import GridLayout, { type Layout, WidthProvider } from "react-grid-layout";
import type { WidgetRef } from "@/server/schemas/widget";
import { KpiTile } from "@/components/widgets/KpiTile";
import { LineChart } from "@/components/widgets/LineChart";
import { DataTable } from "@/components/widgets/DataTable";

const ResponsiveGrid = WidthProvider(GridLayout);

function widgetsToLayout(widgets: WidgetRef[]): Layout[] {
  return widgets.map((w) => ({
    i: w.id,
    x: w.x,
    y: w.y,
    w: w.w,
    h: w.h,
    minW: 1,
    minH: 1,
  }));
}

function applyLayout(widgets: WidgetRef[], layout: Layout[]): WidgetRef[] {
  const byId = new Map(layout.map((l) => [l.i, l]));
  return widgets.map((w) => {
    const l = byId.get(w.id);
    if (!l) return w;
    return { ...w, x: l.x, y: l.y, w: l.w, h: l.h };
  });
}

function WidgetBody({ widget }: { widget: WidgetRef }) {
  switch (widget.kind) {
    case "kpi":
      return <KpiTile widgetId={widget.id} title={widget.title} />;
    case "line":
      return <LineChart widgetId={widget.id} title={widget.title} />;
    case "table":
      return <DataTable widgetId={widget.id} title={widget.title} />;
    default: {
      const _exhaustive: never = widget.kind;
      return <div>Unsupported: {_exhaustive}</div>;
    }
  }
}

interface Props {
  widgets: WidgetRef[];
  onChange: (next: WidgetRef[]) => void;
}

export function EditableDashboardGrid({ widgets, onChange }: Props) {
  const layout = useMemo(() => widgetsToLayout(widgets), [widgets]);
  const [hoveredId, setHoveredId] = useState<string | null>(null);

  return (
    <ResponsiveGrid
      className="layout"
      layout={layout}
      cols={12}
      rowHeight={80}
      margin={[16, 16]}
      draggableCancel=".widget-remove-button"
      onLayoutChange={(next) => onChange(applyLayout(widgets, next))}
      isDraggable
      isResizable
    >
      {widgets.map((w) => (
        <div
          key={w.id}
          data-widget-id={w.id}
          className="relative"
          onMouseEnter={() => setHoveredId(w.id)}
          onMouseLeave={() => setHoveredId(null)}
        >
          {hoveredId === w.id && (
            <button
              type="button"
              aria-label={`Remove ${w.title}`}
              className="widget-remove-button absolute right-2 top-2 z-10 flex h-6 w-6 items-center justify-center rounded-full border border-border bg-card text-sm hover:bg-red-500 hover:text-white"
              onClick={(e) => {
                e.stopPropagation();
                onChange(widgets.filter((x) => x.id !== w.id));
              }}
            >
              ×
            </button>
          )}
          <WidgetBody widget={w} />
        </div>
      ))}
    </ResponsiveGrid>
  );
}
```

```ts
"use client";

import { useMutation, useQueryClient } from "@tanstack/react-query";
import type { Dashboard } from "@/server/schemas/dashboard";
import type { WidgetRef } from "@/server/schemas/widget";

async function saveDashboardApi(id: string, widgets: WidgetRef[]): Promise<Dashboard> {
  const r = await fetch(`/api/dashboards/${encodeURIComponent(id)}`, {
    method: "PUT",
    headers: { "Content-Type": "application/json" },
    credentials: "include",
    body: JSON.stringify({ widgets }),
  });
  if (!r.ok) {
    const text = await r.text();
    throw new Error(`save failed: ${r.status} ${text}`);
  }
  return (await r.json()) as Dashboard;
}

export function useSaveDashboard(id: string) {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (widgets: WidgetRef[]) => saveDashboardApi(id, widgets),
    onSuccess: (saved) => {
      qc.setQueryData(["dashboard", id], saved);
      void qc.invalidateQueries({ queryKey: ["dashboards"] });
    },
  });
}
```

## Task 10: Edit page route

**Files:**
- Create: `app/dashboards/[id]/edit/page.tsx`

- [ ] **Step 1: Write the page**

```tsx
"use client";

import "react-grid-layout/css/styles.css";
import "react-resizable/css/styles.css";

import { useEffect, useMemo, useState, use } from "react";
import { useRouter } from "next/navigation";
import { AppShell } from "@/components/AppShell";
import { EditableDashboardGrid } from "@/components/EditableDashboardGrid";
import { EditToolbar } from "@/components/EditToolbar";
import { useDashboard } from "@/hooks/useDashboard";
import { useSaveDashboard } from "@/hooks/useSaveDashboard";
import type { WidgetRef } from "@/server/schemas/widget";
import type { Route } from "next";

function stableKey(widgets: WidgetRef[]): string {
  return JSON.stringify(
    widgets.map((w) => [w.id, w.kind, w.title, w.x, w.y, w.w, w.h]),
  );
}

export default function DashboardEditPage({
  params,
}: {
  params: Promise<{ id: string }>;
}) {
  const { id } = use(params);
  const router = useRouter();
  const { data, isLoading, error } = useDashboard(id);
  const save = useSaveDashboard(id);

  const [editWidgets, setEditWidgets] = useState<WidgetRef[] | null>(null);

  useEffect(() => {
    if (data && editWidgets === null) {
      setEditWidgets(data.widgets);
    }
  }, [data, editWidgets]);

  const isDirty = useMemo(() => {
    if (!data || editWidgets === null) return false;
    return stableKey(editWidgets) !== stableKey(data.widgets);
  }, [data, editWidgets]);

  const viewHref = `/dashboards/${id}` as Route;

  const handleSave = () => {
    if (editWidgets === null) return;
    save.mutate(editWidgets, {
      onSuccess: () => {
        router.push(viewHref);
      },
    });
  };

  const handleCancel = () => {
    if (isDirty && !window.confirm("Discard unsaved changes?")) return;
    router.push(viewHref);
  };

  return (
    <AppShell>
      {isLoading && <p className="text-muted-foreground">Loading dashboard…</p>}
      {error && (
        <p className="text-red-400">
          Failed to load dashboard: {(error as Error).message}
        </p>
      )}
      {save.error && (
        <p className="text-red-400">
          Save failed: {(save.error as Error).message}
        </p>
      )}
      {data && editWidgets !== null && (
        <>
          <EditToolbar
            title={data.name}
            isDirty={isDirty}
            isSaving={save.isPending}
            onSave={handleSave}
            onCancel={handleCancel}
          />
          <EditableDashboardGrid
            widgets={editWidgets}
            onChange={setEditWidgets}
          />
        </>
      )}
    </AppShell>
  );
}
```

```ts
import { test, expect } from "@playwright/test";

test("edits a dashboard: drag, save, persistence round-trip", async ({ page }) => {
  // Navigate to detail view
  await page.goto("/dashboards/1");
  await expect(page.getByRole("heading", { name: "Infrastructure Overview" })).toBeVisible();

  // Enter edit mode
  await page.getByRole("link", { name: "Edit" }).click();
  await expect(page).toHaveURL(/\/dashboards\/1\/edit$/);
  await expect(page.getByText("no changes")).toBeVisible();

  // Find the grid and the CPU tile's draggable wrapper
  const grid = page.locator(".react-grid-layout");
  await expect(grid).toBeVisible();

  // Drag the KPI tile (data-widget-id="w-cpu-kpi") to a new position.
  // react-grid-layout wraps our child in .react-grid-item, which is the
  // draggable/positioned element.
  const cpuTile = page.locator(
    '.react-grid-item:has([data-widget-id="w-cpu-kpi"])',
  );
  const box = await cpuTile.boundingBox();
  if (!box) throw new Error("CPU tile not found");
  await page.mouse.move(box.x + 20, box.y + 20);
  await page.mouse.down();
  await page.mouse.move(box.x + 400, box.y + 300, { steps: 10 });
  await page.mouse.up();

  // Dirty indicator should appear
  await expect(page.getByText("unsaved changes")).toBeVisible({ timeout: 5_000 });

  // Save
  await page.getByRole("button", { name: /^Save/ }).click();

  // Return to view route
  await expect(page).toHaveURL(/\/dashboards\/1$/, { timeout: 10_000 });
  await expect(page.getByRole("heading", { name: "Infrastructure Overview" })).toBeVisible();

  // Reload and confirm the server saw the PUT (WireMock echo means the GET reflects persisted state only for the current session;
  // we check at least that the Save round-tripped without error by reloading and ensuring no error banner appears).
  await page.reload();
  await expect(page.getByText(/Failed to load/)).not.toBeVisible();
});
```

Caveat: because the WireMock mapping for `GET /dashboards/1.xml` serves the static fixture, the reload will show the original positions, not the dragged ones. The assertion above only verifies the round-trip did not error. A richer persistence check requires a WireMock scenario state machine (record → replay) and is out of scope for P2.1. Document this gap explicitly in the test file as a top-of-file comment.
